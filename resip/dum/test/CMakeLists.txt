FUNCTION(TEST)
   TEST_BASE(${ARGV})
   IF(CMAKE_HOST_WIN32)
      TARGET_LINK_LIBRARIES(${ARGV0} rutil resiprocate dum)
   ELSE()
      TARGET_LINK_LIBRARIES(${ARGV0} rutil resip dum ${POPT_LIBRARY})
   ENDIF()
   SET_TARGET_PROPERTIES(${ARGV0} PROPERTIES FOLDER resip/Tests)   
ENDFUNCTION()
